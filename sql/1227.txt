SELECT       A.PROD_ID, A.PROD_NM,       A.TRACK_ID, A.TRACK_NM,       A.ALBUM_ID, A.ALBUM_NM,       A.ARTIST_ID, A.ARTIST_NM,      A.LABEL_NM, A.GNR_NM,      TO_CHAR(C.RELEASE_DT,'YYYYMMDD') RELEASE_DT FROM PM_PROD_SNG_MASTER_TB A, CM_ALBUM_TB C WHERE EXISTS (SELECT '1'              FROM PM_PROD_SNG_TB B              WHERE B.STATUS_CD = 'LC000030'                AND TO_CHAR(B.MOD_DT,'YYYYMMDD')                    BETWEEN 20070614 AND 20070618 /*Date range for tracks made available */               AND A.PROD_ID = B.PROD_ID            )  AND C.ALBUM_ID = A.ALBUM_ID AND C.LABEL_ID != '7'
